突破

break 语句是用来提前结束一个 for, repeat, while或 do / until某种loop ，或者告诉一个 switch语句在该点结束，或者提前结束一个 with调用。请看这些不同函数的单独页面，以获得更深入的解释，即在每种情况下如何使用它。请注意，如果在这些情况之外使用 break ，将会产生一个错误。

下面你可以看到几个基本的例子，说明如何使用 break ，其语法很简单。

break;

break 在一个 for 循环中。

for (var i = 0; i < 10; i += 1)
{
    if (array[i] == 234)
    {
        break;
    }
}

num = i;

break 在一个 repeat 循环中。

var i = 0;
var temp = 0;
repeat (10)
{
    temp += array[i];
    if (temp > max_total)
    {
        break;
    }
    else
    {
        i += 1;
    }
}

break 在一个 while 循环中。

var i = 0;
while (!place_free(x, y))
{
    x = random(room_width);
    y = random(room_height);
    if (i > 50)
    {
        break;
    }
    else
    {
        i+=1;
    }
}

break 在一个 do / until 循环中。

var _id = noone;
do
{
    _id = list[| 0];
    if instance_exists(_id)
    {
        break;
    }
    ds_list_delete(list, 0);
}
until (ds_list_empty(list));
target = _id;

break ，当使用 with 。

var count = 0;
with (obj_Enemy)
{
    count++;
    if (count > 10)
    {
        break;
    }
    hp = 100;
}

break in a switch:

switch (keyboard_key)
{
    case vk_left:
    case ord("A"):
        x -= 4;
    break;

    case vk_right:
    case ord("D"):
        x += 4;
    break;

    case vk_up:
    case ord("W"):
        y -= 4;
    break;

    case vk_down:
    case ord("S"):
        y += 4;
    break;
}
